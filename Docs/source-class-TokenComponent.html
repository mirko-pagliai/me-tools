 29: /**
 30:  * A component to handle tokens.
 31:  */
 32: class TokenComponent extends Component {
 33:     /**
 34:      * Construct. It loads the Token model.
 35:      * @param ComponentCollection $collection
 36:      * @param array $settings Array of configuration settings
 37:      */
 38:     public function __construct(ComponentCollection $collection, $settings = array()) {
 39:         parent::__construct($collection, $settings);
 40:         $this->Token = ClassRegistry::init('MeTools.Token');
 41:     }
 42:     
 43:     /**
 44:      * Creates a token. Internal function.
 45:      * 
 46:      * If the salt is empty, it will use the current timestamp.
 47:      * @param string $salt Salt to use to generate the token
 48:      * @param int $maxLenght Maximum length of the token
 49:      * @param string $method Method to use (sha1/sha256/md5/blowfish)
 50:      * @return string Token
 51:      */
 52:     private function _createToken($salt = NULL, $maxLenght = 25) {
 53:         $token = Security::hash(empty($salt) ? time() : $salt, 'sha1', TRUE);
 54:         
 55:         //Truncates the token, if it's longer than the maximum length
 56:         $token = strlen($token) > $maxLenght ? substr($token, 0, $maxLenght) : $token;
 57:         
 58:         return $token;
 59:     }
 60:     
 61:     /**
 62:      * Checks if a token exists.
 63:      * 
 64:      * If the user ID is empty or 0, it will check for a token that is not related to a user.
 65:      * @param string $token The token
 66:      * @param string $type Type of the token
 67:      * @param int $user_id User ID, otherwise 0 if the token is not related to a user
 68:      * @return mixed The token ID if the token exists, otherwise FALSE
 69:      */
 70:     public function check($token, $type, $user_id = 0) {
 71:         $conditions = array(
 72:             'expiration >'  => CakeTime::format(time(), '%Y-%m-%d %H:%M:%S'),
 73:             'type'          => $type,
 74:             'value'         => $token
 75:         );
 76:         
 77:         if(!empty($user_id))
 78:             $conditions['user_id'] = $user_id;
 79:         
 80:         $token = $this->Token->find('first', array('conditions' => $conditions, 'fields' => 'id'));
 81:         
 82:         return empty($token['Token']['id']) ? FALSE : $token['Token']['id'];
 83:     }
 84:     
 85:     /**
 86:      * Creates and saves a token.
 87:      * 
 88:      * If the salt is empty, it will use the current timestamp.
 89:      * If the user ID is empty or 0, it will create a token that is not related to a user.
 90:      * If the expiration is empty, will be set to 12 hours.
 91:      * @param string $salt Salt to use to generate the token
 92:      * @param string $type Type of the token
 93:      * @param int $user_id User ID, otherwise 0 if the token is not related to a user
 94:      * @param string $expiration Expiration, strftime compatible formatting
 95:      * @return mixed The token value on success, otherwise FALSE
 96:      * @see http://php.net/strftime strftime documentation
 97:      * @uses _createToken() to create the token
 98:      */
 99:     public function create($salt = NULL, $type, $user_id = 0, $expiration = NULL) {     
100:         $this->Token->create();
101:         $save = $this->Token->save(array(
102:             'expiration'    => CakeTime::format(empty($expiration) ? '+12 hours' : $expiration, '%Y-%m-%d %H:%M:%S'),
103:             'type'          => $type,
104:             'user_id'       => $user_id,
105:             'value'         => $token = self::_createToken($salt)
106:         ));
107:         
108:         return empty($save) ? FALSE : $token;
109:     }
110:     
111:     /**
112:      * Deletes a token
113:      * @param int $id Token id
114:      * @return boolean TRUE on success
115:      */
116:     public function delete($id) {
117:         return $this->Token->delete($id);
118:     }
119: }
